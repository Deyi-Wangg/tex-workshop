 In figure \ref{fig-gantt} you can see an example Gantt chart.
 \begin{figure}
 \centering
    \begin{ganttchart}[hgrid, vgrid]{11}
      \gantttitle{2016}{11} \\
      \gantttitlelist{5,...,15}{1} \\
      \ganttbar{Sample collection}{1}{5} \\
      \ganttbar{Thesis writing}{5}{11} \\
      \ganttgroup{Analysis}{4}{11} \\
      \ganttbar{Preprocessing}{4}{6} \\
      \ganttbar{Classification}{7}{11} \\
      \ganttbar{Presentation}{10}{10} \\
      \ganttmilestone{Midterm}{10}
    \end{ganttchart}
  \caption{Simple example of a Gantt chart, using two titles, five bars, one group and one milestone.}
  \label{fig-gantt}
\end{figure}
